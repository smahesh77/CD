PROGRAM -> MAIN BLOCK
BLOCK -> '{' CODE '}'
CODE -> BLOCK | STATEMENT CODE | STATEMENT
STATEMENT -> DECL ';'
           | ASSIGNMENT ';'
           | CONDITION
           | WHILEST
DECL -> TYPE VARLIST
VARLIST -> VAR ',' VARLIST | VAR
ASSIGNMENT -> VAR '=' EXPR
CONDITION -> IFST | ELSEST
IFST -> IF '(' CONDITION ')' BLOCK
ELSEST -> ELSE BLOCK
WHILEST -> WHILE '(' CONDITION ')' BLOCK
EXPR -> TERM ((ADD | SUB) TERM)*
TERM -> FACTOR ((MUL | DIV) FACTOR)*
FACTOR -> NUM | VAR
